cmake_minimum_required(VERSION 3.19)
project(nvshmem_uid_test LANGUAGES CUDA CXX)

find_package(CUDAToolkit REQUIRED)
find_package(NVSHMEM REQUIRED HINTS ${NVSHMEM_PREFIX}/lib/cmake/nvshmem)

add_executable(nvshmem_uniqueid_test nvshmem_uniqueid_test.cu)
target_link_libraries(nvshmem_uniqueid_test
    PRIVATE nvshmem::nvshmem_host nvshmem::nvshmem_device CUDA::cudart)
target_compile_features(nvshmem_uniqueid_test PRIVATE cxx_std_11 cuda_std_11)
set_target_properties(nvshmem_uniqueid_test PROPERTIES
    POSITION_INDEPENDENT_CODE    ON
    CXX_STANDARD_REQUIRED        ON
    CUDA_STANDARD_REQUIRED       ON
    CXX_STANDARD                 11
    CUDA_STANDARD                11
    CUDA_SEPARABLE_COMPILATION   ON
)
